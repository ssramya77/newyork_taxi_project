SELECT COUNT(*) as total_trips,
                      SUM(total_amount) as total_revenue,
                      AVG(TRIP_DURATION_MINUTES) as avg_trip_duration
                      FROM analytics.dbt_ssimhadri.fct_trips

SELECT DAYNAME(pickup_ts),
        EXTRACT(hour FROM pickup_ts) AS hour,
        COUNT(distinct trip_id) AS trips
    FROM analytics.dbt_ssimhadri.fct_trips
    --WHERE DAYNAME(pickup_ts) = 'Monday'
    GROUP BY 1,2
    ORDER BY 1;


 SELECT
        pu.zone AS pickup_zone,
        do.zone AS dropoff_zone,
        COUNT(*) AS trips
    FROM analytics.dbt_ssimhadri.fct_trips t
    LEFT JOIN analytics.dbt_ssimhadri.dim_zones pu
        ON t.pu_location_id = pu.location_id
    LEFT JOIN analytics.dbt_ssimhadri.dim_zones do
        ON t.do_location_id = do.location_id
    WHERE DAYNAME(t.pickup_ts) = 'Sun'
    GROUP BY 1,2
    ORDER BY trips DESC
    LIMIT 5
